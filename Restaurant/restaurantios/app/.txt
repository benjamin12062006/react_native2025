# CONFIGURACIÓN DOCKER Y GNS3

## Dockerfile
FROM ubuntu:20.04
RUN apt-get update && \
    apt-get install -y python3 python3-pip qemu-kvm libvirt-daemon-system
RUN pip3 install gns3-server
EXPOSE 3080
CMD ["gns3server", "--local"]

## Comandos Docker
docker pull frrouting/frr
docker pull gns3-server:latest
docker stop gns3-server && docker rm gns3-server
docker run … gns3-server:latest

# Shell del servidor GNS3
docker exec -it gns3-server bash

# Shell de un contenedor FRR
docker exec -it <container_id> sh

## Configuración FRR
cd /usr/lib/frr
./watchfrr zebra eigrpd &
vtysh

# CONFIGURACIÓN EIGRP

## Router FRR-1
frrouting-frr-1# conf t
frrouting-frr-1(config)# router eigrp 10
frrouting-frr-1(config-router)# network 192.168.0.0/24
frrouting-frr-1(config-router)# network 10.0.0.0/30
frrouting-frr-1(config-router)# exit
frrouting-frr-1(config)# exit
frrouting-frr-1# wr

## Router FRR-2
